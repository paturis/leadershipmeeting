⚙️ Connect Google Sheet (One-time Setup)

To save updates directly from this page back to your Google Sheet, deploy a small Apps Script as a Web App. This takes about 3 minutes.

1Open your Google Sheet, go to Extensions → Apps Script
2Delete any existing code, paste the script below, and click Save
// SSSIO Tracker — Apps Script Web App // Paste this in Extensions → Apps Script, then Deploy → New deployment const SHEET_NAME = 'Tasks'; // ← change to your sheet tab name if different const TRACKER_URL = 'https://paturis.github.io/leadershipmeeting/'; const TRACKER_PASS = 'SaiRam2026'; function doGet() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const headers = data[0].map(String); const rows = data.slice(1).map((row, i) => { const obj = { _row: i + 2 }; headers.forEach((h, j) => { obj[h] = String(row[j] || ''); }); return obj; }).filter(r => r['Workstream'] && r['Workstream'].trim()); const out = ContentService.createTextOutput(JSON.stringify({ headers, rows })); out.setMimeType(ContentService.MimeType.JSON); return out; } function doPost(e) { const sheet = getSheet(); const payload = JSON.parse(e.postData.contents); const headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0].map(String); if (payload.action === 'update') { Object.entries(payload.data).forEach(([key, val]) => { const col = headers.indexOf(key); if (col >= 0) sheet.getRange(payload._row, col + 1).setValue(val); }); if (payload.notify && payload.data['Owner Email']) { sendEmail(payload.data['Owner Email'], payload.data['Owner'] || '', payload.data['Task'] || '', payload.data); } } else if (payload.action === 'add') { sheet.appendRow(headers.map(h => payload.data[h] || '')); if (payload.notify && payload.data['Owner Email']) { sendEmail(payload.data['Owner Email'], payload.data['Owner'] || '', payload.data['Task'] || '', payload.data); } } const out = ContentService.createTextOutput(JSON.stringify({ ok: true })); out.setMimeType(ContentService.MimeType.JSON); return out; } function sendEmail(toEmail, ownerName, taskName, task) { const subject = '[SSSIO Conference] Task assigned: ' + taskName; const body = 'Hi ' + (ownerName || 'there') + ',\n\n' + 'You have been assigned a task for the SSSIO Leadership Conference (July 24–29, 2026).\n\n' + 'Task: ' + (task['Task'] || taskName) + '\n' + 'Workstream: ' + (task['Workstream'] || '') + '\n' + 'Due Date: ' + (task['Due Date'] || 'TBD') + '\n' + 'Status: ' + (task['Status'] || 'Not Started') + '\n' + (task['Latest Update'] ? 'Notes: ' + task['Latest Update'] + '\n' : '') + '\nTrack and update your tasks at:\n' + TRACKER_URL + '\nPassword: ' + TRACKER_PASS + '\n\nJai Sai Ram 🕉️'; GmailApp.sendEmail(toEmail, subject, body); } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); return ss.getSheetByName(SHEET_NAME) || ss.getSheets()[0]; }
3Click Deploy → New deployment → type: Web app → Execute as: Me → Who has access: Anyone → Deploy
4Copy the Web app URL and paste it below, then click Save

Also make sure your sheet tab is named Tasks (or update SHEET_NAME in the script to match your tab name).
